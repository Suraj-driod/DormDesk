//issues table
create table public.issues (
  id uuid not null default gen_random_uuid (),
  title text not null,
  description text null,
  category public.issue_category not null,
  priority public.issue_priority null default 'low'::issue_priority,
  status public.issue_status null,
  visibility public.issue_visibility_type null default 'public'::issue_visibility_type,
  created_by uuid null,
  assigned_to uuid null,
  hostel text null,
  block text null,
  floor text null,
  room_no text null,
  created_at timestamp with time zone null default now(),
  resolved_at timestamp with time zone null,
  media_url text null,
  repost_count integer null default 0,
  constraint issues_pkey primary key (id),
  constraint issues_assigned_to_fkey foreign KEY (assigned_to) references profile (id) on delete set null,
  constraint issues_created_by_fkey foreign KEY (created_by) references profile (id) on delete CASCADE
) TABLESPACE pg_default;


//profile table
create table public.profile (
  id uuid not null,
  name text not null,
  phone_no integer null,
  email text not null,
  role public.user_role not null,
  hostel text null,
  block text null,
  floor text null,
  room_no text null,
  created_at timestamp with time zone null default now(),
  constraint profile_pkey primary key (id),
  constraint profile_id_fkey foreign KEY (id) references auth.users (id) on delete CASCADE
) TABLESPACE pg_default;


//issue_comments table
create table public.issue_comments (
  id uuid not null default gen_random_uuid (),
  issue_id uuid null,
  user_id uuid null,
  parent_comment_id uuid null,
  content text not null,
  created_at timestamp with time zone null default now(),
  constraint issue_comments_pkey primary key (id),
  constraint issue_comments_issue_id_fkey foreign KEY (issue_id) references issues (id) on delete CASCADE,
  constraint issue_comments_parent_comment_id_fkey foreign KEY (parent_comment_id) references issue_comments (id) on delete CASCADE,
  constraint issue_comments_user_id_fkey foreign KEY (user_id) references profile (id) on delete CASCADE
) TABLESPACE pg_default;


//announcements table
create table public.announcements (
  id uuid not null default gen_random_uuid (),
  title text not null,
  content text not null,
  target_hostel text not null,
  target_block text null,
  target_floor text null,
  created_by uuid null,
  created_at timestamp with time zone null default now(),
  image_url text null,
  poll_data jsonb null,
  constraint announcements_pkey primary key (id),
  constraint announcements_created_by_fkey foreign KEY (created_by) references profile (id) on delete CASCADE
) TABLESPACE pg_default;

//complaints table
create table public.complaints (
  id uuid not null default gen_random_uuid (),
  complaint_type public.complaint_type not null,
  description text not null,
  accused_user text not null,
  raised_by uuid null,
  status public.complaint_status null default 'submitted'::complaint_status,
  created_at timestamp with time zone null default now(),
  closed_at timestamp with time zone null,
  constraint complaints_pkey primary key (id),
  constraint complaints_raised_by_fkey foreign KEY (raised_by) references profile (id) on delete CASCADE
) TABLESPACE pg_default;

//issue_status_logs
create table public.issue_status_logs (
  id uuid not null default gen_random_uuid (),
  issue_id uuid null,
  changed_by uuid null,
  changed_at timestamp with time zone null default now(),
  note text null,
  constraint issue_status_logs_pkey primary key (id),
  constraint issue_status_logs_changed_by_fkey foreign KEY (changed_by) references profile (id) on delete CASCADE,
  constraint issue_status_logs_issue_id_fkey foreign KEY (issue_id) references issues (id) on delete CASCADE
) TABLESPACE pg_default;

//issue_upvotes
create table public.issue_upvotes (
  id uuid not null default gen_random_uuid (),
  issue_id uuid null,
  user_id uuid null,
  created_at timestamp with time zone null default now(),
  constraint issue_upvotes_pkey primary key (id),
  constraint issue_upvotes_issue_id_user_id_key unique (issue_id, user_id),
  constraint issue_upvotes_issue_id_fkey foreign KEY (issue_id) references issues (id) on delete CASCADE,
  constraint issue_upvotes_user_id_fkey foreign KEY (user_id) references profile (id) on delete CASCADE
) TABLESPACE pg_default;

//lost_items table
create table public.lost_items (
  id uuid not null default gen_random_uuid (),
  title text not null,
  description text null,
  location text null,
  image_url text null,
  status public.lost_item_status null default 'lost'::lost_item_status,
  reported_by uuid null,
  claimed_by uuid null,
  claimed_at timestamp with time zone null,
  created_at timestamp with time zone null default now(),
  constraint lost_items_pkey primary key (id),
  constraint lost_items_claimed_by_fkey foreign KEY (claimed_by) references profile (id) on delete set null,
  constraint lost_items_reported_by_fkey foreign KEY (reported_by) references profile (id) on delete CASCADE
) TABLESPACE pg_default;

//management table
create table public.management (
  id uuid not null,
  email text not null,
  full_name text null,
  role text not null,
  is_admin boolean null default false,
  is_caretaker boolean null default false,
  hostel_name text null,
  block_text text null,
  phone text null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  constraint management_pkey primary key (id),
  constraint management_email_key unique (email),
  constraint management_id_fkey foreign KEY (id) references auth.users (id) on delete CASCADE,
  constraint management_role_check check (
    (
      role = any (array['admin'::text, 'caretaker'::text])
    )
  )
) TABLESPACE pg_default;
